Usage Reference

Haven Chart Helpers

// Sparkline
havenSparkline('canvas-id', [data], HAVEN.primary[500]);

// Line chart with zones
havenLineChart('canvas-id', { labels, datasets, referenceValue, zones });

// Bar chart (vertical or horizontal)
havenBarChart('canvas-id', { labels, datasets, horizontal: true });

// Color opacity -- ALWAYS use .replace(), never hex suffix:
HAVEN.primary[500].replace(/, 1\)$/, ', 0.08)')  // correct
HAVEN.primary[500] + '15'                          // wrong
